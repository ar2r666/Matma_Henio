// Prototyp projektu strony do ćwiczenia matematyki dla uczniów 5 klasy // Frontend (HTML, CSS, JavaScript) const htmlContent = `

Matematyka - Przygotowanie do Sprawdzianu

`; // Backend (Node.js, Express, OpenAI API) const express = require('express'); const bodyParser = require('body-parser'); const { Configuration, OpenAIApi } = require('openai'); const app = express(); const port = 3000; app.use(bodyParser.json()); const configuration = new Configuration({ apiKey: 'YOUR_OPENAI_API_KEY', }); const openai = new OpenAIApi(configuration); app.post('/generate-tasks', async (req, res) => { const topic = req.body.topic; const prompt = `Stwórz 10 zadań matematycznych dla ucznia 5 klasy z tematu: ${topic}`; try { const completion = await openai.createCompletion({ model: 'text-davinci-003', prompt: prompt, max_tokens: 150, }); const tasks = completion.data.choices[0].text.split('\n').filter(task => task); res.json({ tasks }); } catch (error) { res.status(500).send('Błąd przy generowaniu zadań'); } }); app.listen(port, () => { console.log(`Aplikacja działa na http://localhost:${port}`); });
